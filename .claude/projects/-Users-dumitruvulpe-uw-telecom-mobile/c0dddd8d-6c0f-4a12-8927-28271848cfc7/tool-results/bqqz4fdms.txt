+		protostrate.WithProducerInterceptor(protostrate_instrumented.NewProducerMetrics()),
+		protostrate.WithProducerInterceptor(producer.NewTokenCorrelationInterceptor()),

+		tpubsub.WithProducerEnvelopeTranslator(),
+	)
+	defer protostrateProd.Close()
+	go func() {
+		if err := protostrateProd.Produce(ctx); err != nil {
+			tlog.WithError(err).Fatal(ctx, "stopped producing messages")
+		}
+	}()
+
+	eventProducer := events.NewProducer(serviceevents.NewServiceeventsProducer(protostrateProd))
+

+	tlog.Info(ctx, "Setup scheduler for handset updates")

+		tbootstrap.WithProtostrateProducer(ctx, protostrateProd, *kafkaTopic),
+		tbootstrap.WithOpsServer(ctx, opsServer),
+	)
+	if err != nil {
+		tlog.WithError(err).Error(ctx, "error returned from error group")
+		os.Exit(1)
+	}

+func expireContractsRunner(store Storage, eventProducer Producer) func(ctx context.Context) {
+	return func(ctx context.Context) {
+		contracts, err := store.FindAndUpdateExpiredContracts(ctx)
+		if err != nil {
+			// The function above returns any successful contracts before an error occurred
+			tlog.WithError(err).
+				WithAttribute("successful_count", fmt.Sprintf("%d", len(contracts))).
+				Error(ctx, "failed to find and update expired contracts")
+		}
+
+		for _, contract := range contracts {
+			var expiresAt *timestamppb.Timestamp
+			if contract.ExpiresAt != nil {
+				expiresAt = timestamppb.New(*contract.ExpiresAt)
+			}
+
+			if err := eventProducer.ProduceMobileHandsetContractExpiredSync(ctx, &serviceevents.MobileHandsetContractExpired{
+				ServiceId:  contract.ServiceID,
+				CustomerId: contract.CustomerID,
+				ContractId: contract.ID.String(),
+				HandsetId:  contract.HandsetID,
+				ExpiresAt:  expiresAt,
+			}); err != nil {
+				tlog.WithError(err).
+					WithAttribute("contract_id", contract.ID.String()).
+					Error(ctx, "failed to produce MobileHandsetContractExpired event")
+			}
+		}
+	}

+func (s storage) FindAndUpdateExpiredContracts(ctx context.Context) ([]models.Contract, error) {
+	filter := bson.M{
+		"status":    models.ContractStatusHandsetOrdered,
+		"expiresAt": bson.M{"$lte": s.nowFunc()},
+	}
+	update := bson.M{
+		"$set": bson.M{
+			"status":    models.ContractStatusExpired,
+			"updatedAt": s.nowFunc(),
+		},
+	}
+	opts := options.FindOneAndUpdate().SetReturnDocument(options.After)
+
+	var expired []models.Contract
+	for {
+		var contract models.Contract
+		err := s.contractsCollection.FindOneAndUpdate(ctx, filter, update, opts).Decode(&contract)
+		if errors.Is(err, mongo.ErrNoDocuments) {
+			break
+		}
+		if err != nil {
+			return expired, fmt.Errorf("could not expire contract: %w", err)
+		}
+		expired = append(expired, contract)
+	}
+
+	return expired, nil
+}

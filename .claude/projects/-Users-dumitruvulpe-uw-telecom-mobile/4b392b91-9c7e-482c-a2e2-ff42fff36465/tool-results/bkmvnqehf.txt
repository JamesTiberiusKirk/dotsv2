+func (r *queryResolver) MobileAvailableHandsets(ctx context.Context, serviceID string, includeOutOfStock *bool) ([]*sharedmodel.MobileHandsetBrand, error) {
+	customerAuth, err := r.authorize(ctx, cerbos.GET_AVAILABLE_HANDSETS)
+	if err != nil {
+		return nil, err
+	}
+
+	includeOOS := includeOutOfStock != nil && *includeOutOfStock
+	resp, err := r.HandsetApiClient.GetAvailableHandsets(ctx, &handsetpb.GetAvailableHandsetsRequest{
+		ServiceId:         serviceID,
+		CustomerId:        customerAuth.AccountNumber,
+		IncludeOutOfStock: includeOOS,
+	})
+	if err != nil {
+		return nil, err
+	}
+
+	grouped := map[string]map[string][]*handsetpb.Handset{}
+	for _, h := range resp.GetHandsets() {
+		if grouped[h.GetManufacturer()] == nil {
+			grouped[h.GetManufacturer()] = map[string][]*handsetpb.Handset{}
+		}
+		m := h.GetManufacturer()
+		grouped[m][h.GetModel()] = append(grouped[m][h.GetModel()], h)
+	}
+
+	brands := make([]*sharedmodel.MobileHandsetBrand, 0, len(grouped))
+	for _, brandName := range sortedKeys(grouped) {
+		phones, err := buildPhones(grouped[brandName])
+		if err != nil {
+			return nil, err
+		}
+		brands = append(brands, &sharedmodel.MobileHandsetBrand{Name: brandName, Phones: phones})
+	}
+	return brands, nil
+}
